    def visitVarDecl(self, ctx) -> None:
        type_str = ctx.typee().getText()
        vars_l = self.visit(ctx.id_l())
        for name in vars_l:
            if name in self._symbol_table:
                raise MiniCInternalError(
                    f"Variable {name} has already been declared")
            else:
                tmp = self.fresh_tmp()
                self._symbol_table[name] = tmp
                if type_str not in ("int", "bool"):
                    raise MiniCUnsupportedError(f"Unsupported type {type_str}")
                # Initialization to 0 or False, both represented with 0
                self.add_statement(
                    RiscV.li(tmp, Operands.Immediate(0)))

    def visitAdditiveExpr(self, ctx) -> Operands.Temporary:
        assert ctx.myop is not None

            self.add_statement(RiscV.add(dest_temp, tmpl, tmpr))

            self.add_statement(RiscV.sub(dest_temp, tmpl, tmpr))

    def visitRelationalExpr(self, ctx) -> Operands.Temporary:
        assert ctx.myop is not None
        c = Condition(ctx.myop.type)
        if self._debug:
            print("relational expression:")
            print(Trees.toStringTree(ctx, [], self._parser))
            print("Condition:", c)
        # BEGIN CUT (codegen)
        tmpl = self.visit(ctx.expr(0))
        tmpr = self.visit(ctx.expr(1))
        dest = self.fresh_tmp()
        end_relational = self.fresh_label('end_relational')
        self.add_statement(
            RiscV.li(dest, Operands.Immediate(0)))
        self.add_statement(
            RiscV.conditional_jump(end_relational, tmpl, c.negate(), tmpr))
        self.add_statement(
            RiscV.li(dest, Operands.Immediate(1)))
        self.add_statement(end_relational)
        return dest


        # END CUT

    def visitMultiplicativeExpr(self, ctx) -> Operands.Temporary:
        assert ctx.myop is not None
        div_by_zero_lbl = self.get_label_div_by_zero()
        # BEGIN CUT (codegen)
        tmpl = self.visit(ctx.expr(0))
        tmpr = self.visit(ctx.expr(1))
        dest_temp = self.fresh_tmp()
        if ctx.myop.type == MiniCParser.MULT:
            self.add_statement(RiscV.mul(dest_temp, tmpl, tmpr))
        elif ctx.myop.type == MiniCParser.DIV:
            self.add_statement(RiscV.conditional_jump(
                div_by_zero_lbl, tmpr, Condition("beq"), Operands.ZERO))
            self.add_statement(RiscV.div(dest_temp, tmpl, tmpr))
        elif ctx.myop.type == MiniCParser.MOD:
            self.add_statement(RiscV.conditional_jump(
                div_by_zero_lbl, tmpr, Condition("beq"), Operands.ZERO))
            self.add_statement(RiscV.rem(dest_temp, tmpl, tmpr))
        else:
            raise MiniCInternalError("Multiplicative expr, but not MUL|DIV|MOD?")
        return dest_temp


        # END CUT

    def visitNotExpr(self, ctx) -> Operands.Temporary:
        # BEGIN CUT (codegen)
        temp = self.visit(ctx.expr())
        dest_temp = self.fresh_tmp()
        # there is no boolean not :-(
        labelneg = self.fresh_label("cond_neg")
        labelend = self.fresh_label("cond_end")
        self.add_statement(RiscV.conditional_jump(
            labelneg, temp,
            Condition("beq"), Operands.ZERO))
        self.add_statement(
            RiscV.li(dest_temp, Operands.Immediate(0)))
        self.add_statement(RiscV.jump(labelend))
        self.add_statement(labelneg)
        self.add_statement(
            RiscV.li(dest_temp, Operands.Immediate(1)))
        self.add_statement(labelend)
        return dest_temp

    def visitFuncDef(self, ctx) -> None:
        funcname = ctx.ID().getText()
        self._current_function = LinearCode(funcname)

        # Our code can deal with multiple functions, and each function is a
        # different LinearCode object. But don't bother the user with
        # self._current_function each time, define a few shortcuts to make the
        # code simpler. This comes here and not in __init__ since we need
        # _current_fuction to be set, and need to redefine the shortcut when
        # _current_function changes.
        self.fresh_tmp = self._current_function.fdata.fresh_tmp
        self.fresh_label = self._current_function.fdata.fresh_label
        self.add_statement = self._current_function.add_statement
        self.add_statement = self._current_function.add_statement
        self.add_instruction_PRINTLN_INT = self._current_function.add_instruction_PRINTLN_INT
        self.add_comment = self._current_function.add_comment
        self.get_label_div_by_zero = self._current_function.fdata.get_label_div_by_zero

        self._symbol_table = dict()
        # BEGIN CUT (functions)
        self.save_regs(Operands.S[1:], 16)
        if ctx.param_l() is None:
            params = []
        else:
            params = self.visit(ctx.param_l())
        i = 0
        for p in params:
            t = self.fresh_tmp()
            self._symbol_table[p] = t
            self.add_statement(RiscV.mv(t, Operands.A[i]))
            i += 1
        # END CUT

        self.visit(ctx.vardecl_l())
        self.visit(ctx.block())
        self.add_comment("Return at end of function:")
        # BEGIN CUT (functions)
        ret_temp = self.visit(ctx.expr())
        self.add_statement(RiscV.mv(Operands.A0, ret_temp))

        # we leave 2 words for the saving of sp and fp
        self.restore_regs(Operands.S[1:], 16)

        # END CUT
        # UNCOMMENT (functions) # This skeleton doesn't deal properly with functions, and
        # UNCOMMENT (functions) # hardcodes a "return 0;" at the end of function. Generate
        # UNCOMMENT (functions) # code for this "return 0;".
        # UNCOMMENT (functions) self.add_statement(
        # UNCOMMENT (functions)     RiscV.li(Operands.A0, Operands.Immediate(0)))
        self._functions.append(self._current_function)
        del self._current_function

    def visitFuncCall(self, ctx) -> Operands.Temporary:
        name = ctx.ID().getText()
        if ctx.expr_l() is None:
            params = []
        else:
            params = self.visit(ctx.expr_l())
        arg_temps = []
        for e in params:
            arg_temps.append(self.visit(e))

        # we leave the space to save both sp and fp, and the registers si
        self.save_regs(Operands.T, (2 + len(Operands.S[1:])) * 8)

        i = 0
        for e in params:
            self.add_statement(RiscV.mv(Operands.A[i], arg_temps[i]))
            i += 1
        self.add_statement(RiscV.call(name))
        self.restore_regs(Operands.T, (2 + len(Operands.S[1:])) * 8)
        dest_temp = self.fresh_tmp()
        self.add_statement(RiscV.mv(dest_temp, Operands.A0))
        return dest_temp

    def visitIfStat(self, ctx) -> None:
        if self._debug:
            print("if statement")
        end_if_label = self.fresh_label("end_if")
        # UNCOMMENT(codegen) raise NotImplementedError()  # TODO
        # BEGIN CUT (codegen)
        else_label = self.fresh_label('else')
        cond = self.visit(ctx.expr())
        self.add_statement(
            RiscV.conditional_jump(
                else_label, cond, Condition("beq"), Operands.ZERO))
        self.visit(ctx.then_block)
        self.add_statement(RiscV.jump(end_if_label))
        self.add_statement(else_label)
        if ctx.else_block is not None:
            if self._debug:
                print("else  ")
            self.visit(ctx.else_block)
        # END CUT
        # BEGIN CUT
        # Following is not needed with RiscV, but was with LEIA.
        # Keeping it here in case we need it later.
        # NOP, to avoid jump with offset +1.
        # self.add_instruction_SUB(R0, R0, 0)
        # END CUT
        self.add_statement(end_if_label)

    def visitWhileStat(self, ctx) -> None:
        if self._debug:
            print("while statement, condition is:")
            print(Trees.toStringTree(ctx.expr(), [], self._parser))
            print("and block is:")
            print(Trees.toStringTree(ctx.stat_block(), [], self._parser))
        # BEGIN CUT (codegen)
        labelbegin = self.fresh_label("begin_while")
        labelend = self.fresh_label("end_while")
        self.add_statement(labelbegin)
        cond_temp = self.visit(ctx.expr())
        self.add_statement(
            RiscV.conditional_jump(
                labelend, cond_temp, Condition("beq"), Operands.ZERO))
        self.visit(ctx.body)
        self.add_statement(RiscV.jump(labelbegin))
        self.add_statement(labelend)
