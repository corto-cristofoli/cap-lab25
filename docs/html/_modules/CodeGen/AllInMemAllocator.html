    def replace(self, old_instr: Instruction) -> List[Instruction]:
        """Replace Temporary operands with the corresponding allocated
        memory location."""
        numreg = 1
        before: List[Instruction] = []
        after: List[Instruction] = []
        new_args: List[Operand] = []
        # UNCOMMENT # TODO: compute before,after,args.
        # UNCOMMENT # TODO: iterate over old_args, check which argument
        # UNCOMMENT # TODO: is a temporary (e.g. isinstance(..., Temporary)),
        # UNCOMMENT # TODO: and if so, generate ld/sd accordingly. Replace the
        # UNCOMMENT # TODO: temporary with S[1], S[2] or S[3] physical registers.
        # BEGIN CUT
        pos = 0
        for arg in old_instr.args():
            new_arg = arg
            if isinstance(arg, Temporary):
                loc = arg.get_alloced_loc()
                if pos == 0 and not old_instr.is_read_only():
                    # Write the content of s2 into address FP+offset*8
                    # destination operand => add store after
                    # We're re-using s2 as destination register, possibly
                    # already as source register. It's harmless (e.g. add
                    # s2, s2, s1), but paranoid people may prefer using
                    # s3, therefore we don't consider s3 as a
                    # general-purpose register.
                    new_arg = S[2]
                    after.append(RiscV.sd(new_arg, loc))
                else:
                    # source operand => add load before
                    new_arg = S[numreg]
                    numreg += 1  # Use a different register for each operand
                    before.append(RiscV.ld(new_arg, loc))
            new_args.append(new_arg)
            pos += 1
        # END CUT
        new_instr = old_instr.with_args(new_args)
        return before + [new_instr] + after

    def prepare(self) -> None:
        """Allocate all temporaries to memory.
        Invariants:
        - Expanded instructions can use s2 and s3
          (to store the values of temporaries before the actual instruction).
        """
        self._fdata._pool.set_temp_allocation(
            {temp: self._fdata.fresh_offset()
             for temp in self._fdata._pool.get_all_temps()})
