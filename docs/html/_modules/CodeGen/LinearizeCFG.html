def ordered_blocks_list(cfg: CFG) -> List[Block]:
    """
    Compute a list of blocks with optimized ordering for linearization.
    """
    # UNCOMMENT # TODO
    # UNCOMMENT return cfg.get_blocks()
    # BEGIN CUT
    visited: Set[Block] = set()
    entries: List[Block] = cfg.get_entries()

    # Depth-first search
    def build_block_list(b: Block) -> List[Block]:
        if b in visited:
            return []
        visited.add(b)
        return sum([build_block_list(out) for out in cfg.out_blocks(b)], [b])

    new_blocks: List[Block] = []
    for e in entries:
        new_blocks += build_block_list(e)
    return new_blocks

def linearize(cfg: CFG) -> List[Statement]:
    """
    Linearize the given control flow graph as a list of instructions.
    """
    # UNCOMMENT # TODO
    l: List[Statement] = []  # Linearized CFG
    blocks: List[Block] = ordered_blocks_list(cfg)  # All blocks of the CFG
    for i, block in enumerate(blocks):
        # 1. Add the label of the block to the linearization
        l.append(block.get_label())
        # 2. Add the body of the block to the linearization
        l.extend(block.get_body())
        # 3. Add the terminator of the block to the linearization
        # BEGIN CUT
        next_label = None if i == len(blocks) - 1 else blocks[i + 1].get_label()
        # END CUT
        match block.get_terminator():
            case BranchingTerminator() as j:
                # BEGIN CUT
                if j.label_else == next_label:  # Do not add the jump to else
                    l.append(ConditionalJump(j.cond, j.op1, j.op2, j.label_then))
                elif j.label_then == next_label:  # Reverse the condition, and do not add the jump
                    l.append(ConditionalJump(j.cond.negate(), j.op1, j.op2, j.label_else))
                else:  # next_label not a target of j: Add the jump
                    l.append(ConditionalJump(j.cond, j.op1, j.op2, j.label_then))
                    l.append(AbsoluteJump(j.label_else))
                # END CUT
                # UNCOMMENT l.append(ConditionalJump(j.cond, j.op1, j.op2, j.label_then))
                # UNCOMMENT l.append(AbsoluteJump(j.label_else))
            case AbsoluteJump() as j:
                # BEGIN CUT
                if j.label != next_label:
                    l.append(AbsoluteJump(j.label))
                # END CUT
                # UNCOMMENT l.append(AbsoluteJump(j.label))
            case Return():
                l.append(AbsoluteJump(cfg.get_end()))
    return l
