def generate_smart_move(dest: DataLocation, src: DataLocation) -> List[BlockInstr]:
    """
    Generate a list of move, store and load instructions, depending on
    whether the operands are registers or memory locations.
    This is an helper function for `sequentialize_moves`.
    """
    instr: List[BlockInstr] = []
    # UNCOMMENT # TODO Compute the moves (Lab 5b, Exercise 4)
    # UNCOMMENT raise NotImplementedError("generate_smart_move")
    # BEGIN CUT
    tmp: Register = S[1]  # S1 is not a general purpose register
    if isinstance(src, Offset) and isinstance(dest, Offset):
        instr = [RiscV.ld(tmp, src), RiscV.sd(tmp, dest)]
    elif isinstance(src, Register) and isinstance(dest, Offset):
        instr = [RiscV.sd(src, dest)]
    elif isinstance(src, Offset) and isinstance(dest, Register):
        instr = [RiscV.ld(dest, src)]
    else:  # Both Register
        instr = [RiscV.mv(dest, src)]
    # END CUT
    return instr

def generate_xor_move(r1: Register, r2: Register) -> List[BlockInstr]:
    """
    Generate a list of xor instructions to swap the two registers,
    without use of a temporary location.
    This is an helper function for `sequentialize_moves`.
    """
    xor1: BlockInstr = RiscV.xor(r2, r1, r2)
    xor2: BlockInstr = RiscV.xor(r1, r1, r2)
    xor3: BlockInstr = RiscV.xor(r2, r1, r2)
    return [xor1, xor2, xor3]

def sequentialize_moves(parallel_moves: Set[Tuple[DataLocation, DataLocation]]
                        ) -> List[BlockInstr]:
    """
    Take a set of parallel moves represented as (destination, source) pairs,
    and return a list of sequential moves which respect the cycles.
    Use the register `tmp` S2 for the cycles.
    Return a corresponding list of RiscV instructions.
    This is an helper function called during SSA exit.
    """
    tmp: Register = S[2]  # S2 is not a general purpose register
    # Build the graph of the moves
    move_graph: DiGraph = DiGraph()
    for dest, src in parallel_moves:
        move_graph.add_edge((src, dest))
    # List for the sequentialized moves to do
    # Convention: in moves we put (dest, src) for each move
    moves: List[Tuple[DataLocation, DataLocation]] = []
    # BEGIN CUT
    # List for the xor to do (xor trick for 2-cycles)
    xor_moves: List[Tuple[Register, Register]] = []
    # END CUT
    # First iteratively remove all the vetices without successors
    vars_without_successor = {src
                              for src, dests in move_graph.neighbourhoods()
                              if len(dests) == 0}
    while vars_without_successor:
        # UNCOMMENT # TODO Remove the leaves iteratively (Lab 5b, Exercise 4)
        # UNCOMMENT raise NotImplementedError("sequentialize_moves: leaves")
        # BEGIN CUT
        var = vars_without_successor.pop()
        for src in move_graph.pred(var):
            moves.append((var, src))
            if len(move_graph.graph_dict[src]) == 1:  # If src had for only predecessor var
                vars_without_successor.add(src)
        move_graph.delete_vertex(var)
        # END CUT
    # Then handle the cycles
    cycles: List = move_graph.connected_components()
    for cycle in cycles:
        # UNCOMMENT # TODO Handle each cycle (Lab 5b, Exercise 4)
        # UNCOMMENT raise NotImplementedError("sequentialize_moves: cycles")
        # BEGIN CUT
        if len(cycle) == 1:
            # Extension part 1
            # Do not add useless mv v, v
            pass
        elif len(cycle) == 2 and isinstance(cycle[0], Register) and isinstance(cycle[1], Register):
            # Extension part 2
            # Xor trick to save a register
            xor_moves.append((cycle[0], cycle[1]))
        else:
            # General case: we need an extra register
            previous = tmp
            for v in reversed(cycle):
                moves.append((previous, v))
                previous = v
            moves.append((previous, tmp))
        # END CUT
    # Transform the moves to do in actual RiscV instructions
    moves_instr: List[BlockInstr] = []
    for dest, src in moves:
        instrs = generate_smart_move(dest, src)
        moves_instr.extend(instrs)
    # BEGIN CUT
    # Transform the xor-moves to do in actual RiscV instructions
    for r1, r2 in xor_moves:
        instrs = generate_xor_move(r1, r2)
        moves_instr.extend(instrs)
    # END CUT
    return moves_instr
