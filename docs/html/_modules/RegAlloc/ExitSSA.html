# Works with naive and all-in-mem allocators, but not the smart one.

def generate_moves_from_phis_naive(phis: List[PhiNode],
                                   parent: Block) -> List[BlockInstr]:  # pragma: no cover
    moves: List[BlockInstr] = []
    for phi in phis:
        src = phi.get_srcs().get(parent.get_label(), None)
        if src:
            i = RiscV.mv(phi.defined()[0], src)
            moves.append(i)
    return moves

def generate_moves_from_phis(phis: List[PhiNode], parent: Block) -> List[BlockInstr]:
    """
    `generate_moves_from_phis(phis, parent)` builds a list of move instructions
    to be inserted in a new block between `parent` and the block with phi nodes
    `phis`.

    This is an helper function called during SSA exit.
    """
    moves: List[BlockInstr] = []
    # UNCOMMENT # TODO compute 'moves', a list of 'mv' instructions to insert under parent
    # UNCOMMENT # (Lab 5a, Exercise 6)
    # BEGIN CUT
    lbl = parent.get_label()
    parallel_moves = set()
    for phi in phis:
        if phi.get_srcs().get(lbl):
            dest = phi.defined()[0]
            src = phi.get_srcs()[lbl]
            # Find new source and destinations
            if isinstance(dest, Temporary):
                dest = dest.get_alloced_loc()  # Register or Offset
            if isinstance(src, Temporary):
                src = src.get_alloced_loc()  # Register or Offset
            parallel_moves.add((dest, src))
    moves = sequentialize_moves(parallel_moves)
    # END CUT
    return moves

def exit_ssa(cfg: CFG, is_smart: bool) -> None:
    """
    `exit_ssa(cfg)` replaces phi nodes with move instructions to exit SSA form.

    `is_smart` is set to true when smart register allocation is enabled (Lab 5b).
    """
    for b in cfg.get_blocks():
        phis = cast(List[PhiNode], b.get_phis())  # Use cast for Pyright
        b.remove_all_phis()  # Remove all phi nodes in the block
        parents: List[Block] = b.get_in().copy()  # Copy as we modify it by adding blocks
        for parent in parents:
            moves = generate_moves_from_phis(phis, parent)
            # UNCOMMENT # TODO Add the block containing 'moves' to 'cfg'
            # UNCOMMENT # and update edges and jumps accordingly (Lab 5a, Exercise 6)
            # UNCOMMENT raise NotImplementedError("exit_ssa")
            # BEGIN CUT
            new_b = Block(cfg.fdata.fresh_label("mv"), moves,
                          AbsoluteJump(b.get_label()))
            cfg.add_block(new_b)
            cfg.remove_edge(parent, b)
            cfg.add_edge(new_b, b)
            match parent.get_terminator():
                case AbsoluteJump() as j:
                    j.label = new_b.get_label()
                case BranchingTerminator() as j:
                    if b.get_label() == j.label_then:
                        j.label_then = new_b.get_label()
                    elif b.get_label() == j.label_else:
                        j.label_else = new_b.get_label()
                case Return():  # pragma: no cover
                    pass  # This case should never happen
            cfg.add_edge(parent, new_b)
